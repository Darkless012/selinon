        if self.is_connected():
            self.conn.connection_pool.disconnect()

[docs]    def retrieve(self, task_name, task_id):
        assert(self.is_connected())

        ret = self.conn.get(task_id)

        if ret is None:
            raise FileNotFoundError("Record not found in database")

        record = json.loads(ret.decode(self.charset))

        assert(record.get('task_name') == task_name)
        return record.get('result')



[docs]    def store(self, node_args, flow_name, task_name, task_id, result):
        assert(self.is_connected())

        record = {
            'node_args': node_args,
            'flow_name': flow_name,
            'task_name': task_name,
            'task_id': task_id,
            'result': result
        }

        self.conn.set(task_id, json.dumps(record))
        return task_id
